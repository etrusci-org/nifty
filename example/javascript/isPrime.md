## Usage

```javascript
import { isPrime } from "./isPrime.js";

let test1 = isPrime(2);
let test2 = isPrime(10);
let test3 = isPrime(1_333_331);

console.log(test1);
// expected output:
// true

console.log(test2);
// expected output:
// false

console.log(test3);
// expected output:
// true
```
